// ========================================== // FOREX Road to 3000 — Google Apps Script // อัปเดต: แก้ column mapping ให้ตรงกับ Sheet จริง // ========================================== const SHEET_NAMES = { q2: 'Q2 2026', q1: 'Q1 2026' }; // Q2 2026 column map const COL_Q2 = { no:1, dateOpen:2, dateClose:3, dur:4, sym:5, pos:6, lot:7, order:8, enter:9, session:10, exit:11, tpsl:12, pnlPct:13, pnlUsd:14, grade:15, setup:16, chartEnter:18, chartExit:19, wl:24, gradeFinal:25, note:26 }; // Q1 2026 column map (W/L=Y=25, Grade=Z=26, Note=AA=27) const COL_Q1 = { no:1, dateOpen:2, dateClose:3, dur:4, sym:5, pos:6, lot:7, order:8, enter:9, session:10, exit:11, tpsl:12, pnlPct:13, pnlUsd:14, grade:15, setup:16, chartEnter:18, chartExit:19, wl:25, gradeFinal:26, note:27 }; const DATA_START_ROW = 8; function doGet(e) { const action = e.parameter.action; const quarter = e.parameter.quarter || 'q2'; let result; if (action === 'getTrades') result = getTrades(quarter); else if (action === 'getAccount') result = getAccountSummary(quarter); else result = { error: 'Unknown action' }; return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const body = JSON.parse(e.postData.contents); const action = body.action; const quarter = body.quarter || 'q2'; let result; if (action === 'updateTrade') result = updateTrade(quarter, body.rowIndex, body.field, body.value); else if (action === 'addTrade') result = addTrade(quarter, body.trade); else if (action === 'deleteTrade') result = deleteTrade(quarter, body.rowIndex); else if (action === 'saveAllTrades') result = saveAllTrades(quarter, body.trades); else result = { error: 'Unknown action' }; return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } function getTrades(quarter) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAMES[quarter]); if (!sheet) return { error: 'ไม่พบ Sheet: ' + SHEET_NAMES[quarter] }; const COL = quarter === 'q1' ? COL_Q1 : COL_Q2; const lastRow = sheet.getLastRow(); const numRows = lastRow - DATA_START_ROW + 1; if (numRows < 1) return { success: true, trades: [], total: 0 }; const allData = sheet.getRange(DATA_START_ROW, 1, numRows, 30).getValues(); const trades = []; allData.forEach((row, i) => { const noVal = row[COL.no - 1]; if (!noVal || isNaN(parseInt(noVal))) return; trades.push({ no: parseInt(noVal), dateOpen: formatDate(row[COL.dateOpen - 1]), dateClose: formatDate(row[COL.dateClose - 1]), dur: str(row[COL.dur - 1]), sym: str(row[COL.sym - 1]), pos: str(row[COL.pos - 1]), lot: str(row[COL.lot - 1]), order: str(row[COL.order - 1]), enter: str(row[COL.enter - 1]), session: str(row[COL.session - 1]), exit: str(row[COL.exit - 1]), tpsl: str(row[COL.tpsl - 1]), pnlPct: Math.round(num(row[COL.pnlPct - 1]) * 10000) / 100, pnlUsd: Math.round(num(row[COL.pnlUsd - 1]) * 100) / 100, grade: str(row[COL.grade - 1]), setup: str(row[COL.setup - 1]).substring(0, 100), chartEnter: extractUrl(row[COL.chartEnter - 1]), chartExit: extractUrl(row[COL.chartExit - 1]), wl: str(row[COL.wl - 1]), gradeFinal: str(row[COL.gradeFinal - 1]), note: str(row[COL.note - 1]).substring(0, 120), _row: DATA_START_ROW + i }); }); return { success: true, trades: trades, total: trades.length }; } catch (err) { return { error: err.toString() }; } } function getAccountSummary(quarter) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAMES[quarter]); const col = quarter === 'q2' ? 30 : 31; return { success: true, summary: { startBal: num(sheet.getRange(4, col).getValue()), withdraw: num(sheet.getRange(5, col).getValue()), curBal: num(sheet.getRange(6, col).getValue()), } }; } catch (err) { return { error: err.toString() }; } } function updateTrade(quarter, actualRow, field, value) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAMES[quarter]); const COL = quarter === 'q1' ? COL_Q1 : COL_Q2; const col = COL[field]; if (!col) return { error: 'ไม่รู้จัก field: ' + field }; let saveValue = value; if (field === 'pnlPct') saveValue = parseFloat(value) / 100; sheet.getRange(parseInt(actualRow), col).setValue(saveValue); return { success: true }; } catch (err) { return { error: err.toString() }; } } function addTrade(quarter, trade) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAMES[quarter]); const COL = quarter === 'q1' ? COL_Q1 : COL_Q2; const newRow = sheet.getLastRow() + 1; Object.keys(COL).forEach(f => { let val = trade[f] !== undefined ? trade[f] : ''; if (f === 'pnlPct') val = (parseFloat(trade.pnlPct) || 0) / 100; sheet.getRange(newRow, COL[f]).setValue(val); }); return { success: true, newRow: newRow }; } catch (err) { return { error: err.toString() }; } } function deleteTrade(quarter, actualRow) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAMES[quarter]); sheet.deleteRow(parseInt(actualRow)); return { success: true }; } catch (err) { return { error: err.toString() }; } } function saveAllTrades(quarter, trades) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAMES[quarter]); const COL = quarter === 'q1' ? COL_Q1 : COL_Q2; let updated = 0; trades.forEach(trade => { if (!trade._row) return; const r = parseInt(trade._row); Object.keys(COL).forEach(f => { if (f === '_row') return; let val = trade[f] !== undefined ? trade[f] : ''; if (f === 'pnlPct') val = (parseFloat(trade.pnlPct) || 0) / 100; sheet.getRange(r, COL[f]).setValue(val); }); updated++; }); return { success: true, updated: updated }; } catch (err) { return { error: err.toString() }; } } function str(v) { if (v === null || v === undefined) return ''; if (v instanceof Date) return formatDate(v); return String(v).trim(); } function num(v) { if (!v && v !== 0) return 0; return parseFloat(v) || 0; } function formatDate(v) { if (!v) return ''; if (v instanceof Date) { return Utilities.formatDate(v, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } const s = String(v).trim(); const match = s.match(/(\d{4}-\d{2}-\d{2})/); if (match) return match[1]; return s; } function extractUrl(v) { if (!v) return ''; const s = String(v).trim(); if (s.startsWith('http')) return s; const match = s.match(/https?:\/\/[^\s"')]+/); return match ? match[0] : ''; }
